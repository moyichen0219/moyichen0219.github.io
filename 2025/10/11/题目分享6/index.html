//高精度减法

#include<bits/stdc++.h>
using namespace std;

const int N = 1e4 +10;
int a1[N] = {0}, a2[N] = {0}, target[N] = {0};

int main(){
string s1, s2;
cin >> s1 >> s2;
char flag = ‘+’;
if (s1.size() < s2.size() || s1.size() == s2.size() && s1 < s2){
swap (s1, s2);
flag = ‘-‘;
}
for (int i = 0; i < s1.size(); i++){
a1[i] = s1[s1.size() - 1 -i] - ‘0’;
}
for (int i= 0; i < s2.size(); i++){
a2[i] = s2[s2.size() - 1 -i] - ‘0’;
}
for (int i = 0; i < s1.size(); i++){
if (a1[i] < a2[i]){
a1[i] += 10;
a1[i+1] -= 1;
}
target[i] = a1[i] - a2[i];
}
if (flag == ‘-‘){
printf(“-“);
}
int index = 0;
for (int i = s1.size() - 1; i >= 0; i–){
if (target[i] != 0){
index = i;
break;
}
}
for (int i = index; i >= 0; i–){
cout << target[i];
}
return 0;
}
